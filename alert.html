            // Prompt the user to enter values
            var num1 = prompt("Enter first value:");
            var num2 = prompt("Enter second value:");
            var opt = prompt("Enter the Operator: ");

            // Convert the string inputs to numbers
            num1 = Number(num1);
            num2 = Number(num2);

            if(opt == '/'){
                // Calculate the div
                var div = num1 / num2;
                // Display the result
                alert("Division is: " , div);
            }

            if(opt == '-'){
                // Calculate the sub
                var sub = num1 - num2;
                // Display the result
                alert("Subtraction is: " , sub);
            }

            if(opt == '*'){
                // Calculate the mul
                var mul = num1 * num2;
                // Display the result
                alert("Multiplication is: " , mul);
            }

            if(opt == '+'){
                // Calculate the sum
                var sum = num1 + num2;
                // Display the result
                alert("SUM: " , sum);
            }
